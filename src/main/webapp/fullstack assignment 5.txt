Working with pre-defined modules of Node JS:

import fs from "fs";


fs.writeFileSync("sync.txt", "This is written synchronously");
const syncData = fs.readFileSync("sync.txt", "utf-8");
console.log("Sync Read:", syncData);


fs.writeFile("async.txt", "This is written asynchronously", (err) => {
  if (err) throw err;
  fs.readFile("async.txt", "utf-8", (err, data) => {
    if (err) throw err;
    console.log("Async Read:", data);
  });
});
